    def initialize(args, &block)
      super
      t = define_outer_task
      t.prerequisites << player_task_name
      
      out_task = mxmlc output do |t|
        configure_mxmlc t
        configure_mxmlc_application t
        t.optimize            = true
        t.warnings            = false
        t.verbose_stacktraces = false
        
        block.call t if !block.nil?
      end
      
      task player_task_name => output
      return out_task
    end
