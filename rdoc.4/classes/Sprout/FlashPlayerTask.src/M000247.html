    def focus_player_on_mac(out, exe)
      return if do_not_focus
      begin
        if(User.new.is_a?(OSXUser))
          require 'appscript'
          Appscript.app(exe).activate
        end
      rescue LoadError => e
        puts "[WARNING] Cannot focus Flash Player without rb-appscript gem"
      end
    end
