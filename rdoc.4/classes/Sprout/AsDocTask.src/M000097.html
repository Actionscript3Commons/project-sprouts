    def prepare
      super
      prerequisite = nil
      @prerequisites.each do |req|
        prerequisite = @application[req]
        if(prerequisite.is_a?(MXMLCTask))
          prerequisite.source_path.each do |path|
            doc_sources << path
          end
          prerequisite.library_path.each do |path|
            library_path << path
          end
        end
      end
    end
