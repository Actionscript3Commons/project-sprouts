    def User.new(os=nil, impl=nil)
        if(os.nil? && impl.nil? && @@user)
          return @@user
        end
        if(os.nil?)
          os = Platform::OS
        end
        if(impl.nil?)
          impl = Platform::IMPL
        end
        if(os == :win32 && impl == :vista)
          @@user = VistaUser.new
        elsif(os == :win32 && impl == :cygwin)
          @@user = CygwinUser.new
        elsif(os == :win32)
          @@user = WinUser.new
        elsif(os == :unix && impl == :macosx)
          @@user = OSXUser.new
#        elsif(os == :unix && impl == :linux)
#          @@user = UnixUser.new
        else
          @@user = UnixUser.new
        end
    end
