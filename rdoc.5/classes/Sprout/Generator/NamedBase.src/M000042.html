      def generate_script_path
        usr = User.new
        if(usr.is_a?(WinUser) && !usr.is_a?(CygwinUser))
          return File.join(class_name, 'script', "generate.rb")
        else
          return File.join(class_name, 'script', "generate")
        end
      end
