    def self.get_executable_from_path(exe)
      path = ENV['PATH']
      file_path = nil
      path.split(get_path_delimiter).each do |p|
        file_path = File.join(p, exe)
#        file_path = file_path.split("/").join("\\")
#        file_path = file_path.split("\\").join("/")
        if(File.exists?(file_path))
          return User.clean_path(file_path)
        end
      end
      return nil
    end
